### Install as package

<RefSubLayout.EducationRow>
  <RefSubLayout.Details>

    You can install @supabase/biobase-js via the terminal.

  </RefSubLayout.Details>

  <RefSubLayout.Examples>

    <Tabs
      size="small"
      type="underlined"
      defaultActiveId="npm"
      queryGroup="platform"
    >
      <TabPanel id="npm" label="npm">

        ```sh Terminal
        npm install @supabase/biobase-js
        ```

      </TabPanel>
      <TabPanel id="yarn" label="Yarn">

        ```sh Terminal
        yarn add @supabase/biobase-js
        ```

      </TabPanel>
      <TabPanel id="pnpm" label="pnpm">

        ```sh Terminal
        pnpm add @supabase/biobase-js
        ```

      </TabPanel>

  </RefSubLayout.Examples>

### Install via CDN

<RefSubLayout.EducationRow>
  <RefSubLayout.Details>
    
    You can install @supabase/biobase-js via CDN links.

  </RefSubLayout.Details>

  <RefSubLayout.Examples>

    ```js
    <script src="https://cdn.jsdelivr.net/npm/@supabase/biobase-js@2"></script>
    //or
    <script src="https://unpkg.com/@supabase/biobase-js@2"></script>
    ```

  </RefSubLayout.Examples>
</RefSubLayout.EducationRow>

### Use at runtime in Deno

<RefSubLayout.EducationRow>
  <RefSubLayout.Details>
    
    You can use biobase-js in the Deno runtime via [JSR](https://jsr.io/@supabase/biobase-js):

  </RefSubLayout.Details>

  <RefSubLayout.Examples>

    ```ts
      import { createClient } from 'jsr:@supabase/biobase-js@2'
    ```

  </RefSubLayout.Examples>
</RefSubLayout.EducationRow>
